Project Overview

This project is a smart energy-saving lighting system built using an ESP32 microcontroller, integrating motion detection, ambient light sensing, and current monitoring to automate and optimize bulb usage.

It connects to Wi-Fi and communicates with Ubidots to publish real-time data such as light status, current consumption, and energy saved, while also allowing remote manual control through a subscribed feed. The system uses a PIR sensor to detect motion, an LDR to measure ambient light, and an ACS712 sensor to monitor current flow.

3. Main Control Loop

void loop() {
  int motion = digitalRead(PIRPIN);
  int lightLevel = analogRead(LDRPIN);
  float current = readCurrent();
  
  if (forceOn) {
    for (int i = 0; i < 4; i++) {
      if (!bulbOn[i]) {
        controlBulb(i, true);
      }
    }
  } else {
    if (motion == HIGH && lightLevel < LIGHT_THRESHOLD) {
      for (int i = 0; i < 4; i++) {
        if (!bulbOn[i]) {
          controlBulb(i, true);
        }
      }
    } else {
      for (int i = 0; i < 4; i++) {
        if (bulbOn[i]) {
          controlBulb(i, false);
        }
      }
    }
  }
  
  delay(500);
}

4. Custom Functions

void controlBulb(int index, bool state) {
  int pins[] = {RELAY1_PIN, RELAY2_PIN, RELAY3_PIN, RELAY4_PIN};
  digitalWrite(pins[index], state ? HIGH : LOW);
  bulbOn[index] = state;
}

float readCurrent() {
  int raw = analogRead(CURRENTSENSORPIN);
  float voltage = (raw / 4095.0) * 3.3;
  float offset = 3.3 / 2.0;
  float sensitivity = 0.185;
  float current = (voltage - offset) / sensitivity;
  return abs(current);
}
